모바일 웹 에디터 미리보기 규약

ex1).
	<form name="form" method="post" action="http://www.munjanote.com/urlplus/preview/">
		<input type="hidden" name="client_id" value="아이디입력.." /><!-- [필수] 사용자 아이디 -->
		<input type="hidden" name="htmlKey" value="" /><!-- [필수] 페이지키 -->
		<input type="submit" value="페이지 호출" />
	</form>

ex2).
	<button onclick="preview()">팝업 호출</button>
	<script type="text/javascript">
	function preview() {
		var f = document.form;
		var url = "http://www.munjanote.com/urlplus/preview/";
		var param = "?client_id=" + f.client_id.value + "&htmlKey=" + f.htmlKey.value;
		window.open(url+param,"urlplusPreview","scrollbars=auto,toolbar=no,resizable=yes,width=400,height=500,left=0,top=0");
	}
	</script>
